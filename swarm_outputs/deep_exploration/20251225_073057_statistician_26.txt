import pandas as pd
import numpy as np

def calculate_gdi(k_values, lambda_val=2.01):
    gdi = []
    for i in range(1, len(k_values)):
        expected_growth = k_values[i-1] * lambda_val
        deviation = (k_values[i] - expected_growth) / expected_growth
        gdi.append(deviation)
    return gdi

# Load k values from dataset

# Segregate k values by GDI threshold (e.g., 0.1)

# Analyze growth patterns in low_gdi_k_values subset
